To learn SQL
# Start server
`docker compose up -d`
# Dive into the docker
`docker exec -it learn-sql bash`
# Go to sql terminal
`mysql -h localhost -P 3306 -u root -pchangeme`

# Stop server
`docker compose stop`
# Stop server and destroy
`docker compose down`
